

```bash

fluorite-mcp --apply-spike nextjs-auth-nextauth-credentials --params '{"app_name":"MyApp"}'

# Preview template before applying
fluorite-mcp --preview-spike fastapi-jwt-auth --params '{"project_name":"MyAPI"}'
```
